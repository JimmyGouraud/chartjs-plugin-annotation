# Bar Chart

const config = {
  type: 'bar',
  data,
  options: {
    plugins: {
      annotation: {
        annotations: {
          annotation1,
          annotation2,
          annotation3
        }
      }
    },
  }
};
const annotation1 = {
  type: 'line',
  scaleID: 'x',
  borderWidth: 3,
  borderColor: 'black',
  value: 0.5,
  label: {
    content: 'Line annotation at x=0.5',
    display: true
  },
};
const annotation2 = {
  type: 'line',
  scaleID: 'x',
  borderWidth: 3,
  borderColor: 'black',
  value: 'Label 5',
  label: {
    rotation: 'auto',
    position: 'start',
    backgroundColor: 'black',
    content: 'Line at x=Label 5',
    display: true
  }
};
const annotation3 = {
  type: 'box',
  xMin: 2.5,
  xMax: 3.5,
  yMin: 0,
  yMax: 100,
  backgroundColor: 'rgba(250, 250, 0, 0.4)',
  borderColor: 'rgba(0, 150, 0, 0.2)',
  drawTime: 'beforeDatasetsDraw',
  borderWidth: 0,
  borderRadius: 0,
};
const DATA_COUNT = 8;
const MIN = 10;
const MAX = 100;
Utils.srand(8);
const labels = [];
for (let i = 0; i < DATA_COUNT; ++i) {
  labels.push('Label ' + i);
}
const numberCfg = {count: DATA_COUNT, min: MIN, max: MAX};
const data = {
  labels: labels,
  datasets: [{
    data: Utils.numbers(numberCfg),
  }, {
    data: Utils.numbers(numberCfg),
  }, {
    data: Utils.numbers(numberCfg),
  }]
};
